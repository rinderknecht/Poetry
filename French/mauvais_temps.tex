\poemtitle{Mauvais temps}

\settowidth{\versewidth}{où je t'ai réchauffée}

\bigskip

\begin{verse}[\versewidth]
  Dans cette valse \\
  de l'oubli \\
  et des regrets

  tu fais un pas \\
  et tu oublies

  j'en fais un autre \\
  et je regrette

  L'un contre l'autre \\
  cherchons \\
  le troisième temps

  comme cet après-midi \\
  de mauvais temps \\
  où je t'ai réchauffée \\
  dans mes bras
\end{verse}
