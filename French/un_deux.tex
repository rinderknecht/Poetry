\poemtitle{Un deux}

\settowidth{\versewidth}{sur les lignes de paume}

\bigskip

\begin{verse}[\versewidth]
  Ta main ballante \\
  abandonnée au vent \\
  comme une liane

  bat doucement

  au rythme des pas \\
  la mesure déliée \\
  de tes cils

  Ma main hésitante \\
  frôle à contretemps \\
  ce pendule vivant

  compte discrètement

  la quinte parfaite \\
  du bout des doigts

  compose secrètement

  une marche \\
  à portée de main
\end{verse}
