\poemtitle{8/9}

\settowidth{\versewidth}{nous partageons un verre}

\bigskip

\begin{verse}[\versewidth]
  Parmi les invités riant \\
  nous partageons un verre \\
  avec deux glaçons

  accolés

  Ces icebergs fatals \\
  complices et beaux \\
  flottent \\
  dans un monde de silence

  donnant aux soleils \\
  le change \\
  au neuvième
\end{verse}
