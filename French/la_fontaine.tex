\poemtitle{La fontaine}

\settowidth{\versewidth}{fait pâlir le soleil}

\bigskip

\begin{verse}[\versewidth]
  Assoiffé

  tu guides mes mains \\
  vers l'eau claire \\
  de la fontaine

  Aveuglé

  de tes lèvres \\
  la goulée \\
  éclipse le soleil
\end{verse}
