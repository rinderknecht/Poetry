\poemtitle{Les bocaux}

\settowidth{\versewidth}{de cette intimité}

\bigskip

\begin{verse}[\versewidth]
  Tu me tends \\
  un second bocal

  Sans réfléchir \\
  je l'ouvre \\
  et tu souris \\
  de profil \\
  de cette intimité

  comme des mariés

  nous sommes \\
  comme ces bocaux \\
  à ouvrir

  sans réfléchir
\end{verse}
