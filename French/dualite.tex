\poemtitle{Dualité}

\settowidth{\versewidth}{d'onde et de particules}

\bigskip

\begin{verse}[\versewidth]
  Espiègle \\
  tu me parles \\
  d'ondes et de particules

  éclectiques

  d'avance de phase \\
  et de décharges

  électriques

  Moi je ne suis que \\
  gravité

  faible \\
  à la portée \\
  de tes bras
\end{verse}
