\poemtitle{La comète}

\settowidth{\versewidth}{Parfois l'île vient au marin}

\bigskip

\begin{verse}[\versewidth]
  Tu me dis que tu es revenue \\
  vers moi

  mais je suis la comète \\
  eccentrique

  De mon archipel \\
  de solitude \\
  loin de ton orbite \\
  circulaire \\
  j'ai senti ton rappel \\
  et fui

  à la vitesse de libération \\
  celle de la lumière \\
  de tes yeux clairs

  Parfois l'île vient au marin
\end{verse}
