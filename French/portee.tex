\poemtitle{Un deux}

\settowidth{\versewidth}{de la canopée et du soleil}

\bigskip

\begin{verse}[\versewidth]
  Une main ballante \\
  abandonnée au vent \\
  comme une liane

  bat doucement

  au rythme de nos pas \\
  la mesure déliée \\
  de tes yeux

  Ce pendule vivant \\
  témoin imparfait \\
  de la canopée et du soleil \\
  qui caresse ta main

  compte discrètement

  la quinte parfaite \\
  du bout des doigts

  compose secrètement

  une marche \\
  sur les lignes de paume \\
  à la portée de ma main
\end{verse}
