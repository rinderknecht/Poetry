\poemtitle{Chevelure}

\settowidth{\versewidth}{d'une lumière chaude}

\bigskip

\begin{verse}[\versewidth]
  Au loin

  les bouleaux \\
  baignés \\
  d'une lumière chaude

  Au près

  tes doigts blancs \\
  peignant \\
  ta chevelure dorée
\end{verse}
