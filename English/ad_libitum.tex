\settowidth{\versewidth}{When you go through my lines}

\bigskip

\begin{verse}[\versewidth]
  When you go through my lines \\
  with me

  sometimes \\
  if slip my own words \\
  and you also want \\
  to slip too

  with me

  a blank page
\end{verse}
