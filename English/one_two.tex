\poemtitle{}

\settowidth{\versewidth}{up to the fluttering measure}

\bigskip

\begin{verse}[\versewidth]
  Her dangling hand \\
  abandoned to the wind \\
  like a liana

  slowly beats \\
  to the rhythm of her steps \\
  up to the fluttering measure \\
  of her eyelashes

  My hesitant hand \\
  brushes off-beat \\
  the divine pendulum

  discreetly counts \\
  perfect fifths \\
  on the tip of my fingers

  secretly composes \\
  a march on a staff
  \end{verse}
