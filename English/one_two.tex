\poemtitle{}

\settowidth{\versewidth}{to the rhythm of her steps}

\bigskip

\begin{verse}[\versewidth]
  Her dangling arms \\
  abandoned to the wind \\
  like lianas

  gently beat \\
  to the rhythm of her steps

  The choir of the forest \\
  around us \\
  murmurs while staring \\
  at the divine pendulums

  waiting for a sign

  Hear the deep chest \\
  of the wild deer \\
  darkly drumming \\
  its untamed song
\end{verse}
