\poemtitle{}

\settowidth{\versewidth}{to the rhythm of her steps}

\bigskip

\begin{verse}[\versewidth]
  Her dangling arms \\
  abandoned to the wind \\
  like lianas

  gently beat \\
  to the rhythm of her steps

  The choir of trees \\
  around us \\
  murmurs while staring \\
  at the divine pendulums

  waiting for a sign

  Listen

  Hear the deer \\
  and its deep chest \\
  darkly drumming \\
  an untamed song
  \end{verse}
