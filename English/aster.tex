\poemtitle{}

\settowidth{\versewidth}{where I seek my reflection}

\bigskip

\begin{verse}[\versewidth]
  In the hills in the east \\
  before sunrise

  I almost passed \\
  a blue aster \\
  in a field of weeds

  a star fallen \\
  with the night rain

  Piercing through \\
  the lifting mist \\
  the magnificent flower \\
  was shivering \\
  under the northern breeze

  Two clear eyes \\
  under the unborn sky

  Two cold hands \\
  under a clearer sky

  whose rising sun \\
  leaves two small puddles \\
  where I seek my reflection \\
  amid the dew
\end{verse}
