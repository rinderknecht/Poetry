\poemtitle{}

\settowidth{\versewidth}{Otherwordly blue sapphire}

\bigskip

\begin{verse}[\versewidth]
  On the eastern plain \\
  before sunrise

  I almost passed \\
  a blue aster \\
  in a field of weeds

  a star fallen \\
  with last night's rain

  Awakening through \\
  the lifting mist \\
  the magnificent flower \\
  seeks earthly warmth \\
  in the northern breeze

  Otherwordly blue sapphire \\
  under sky unborn \\
  oh picking you up \\
  to my beating heart!

  If only the cold wind \\
  had not stolen my hands \\
  under skies clearer \\
  and rising sun

  revealing puddles \\
  with my reflection

  amid the dew
\end{verse}
