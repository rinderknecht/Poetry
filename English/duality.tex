\poemtitle{}

\settowidth{\versewidth}{d'onde et de particules}

\bigskip

\begin{verse}[\versewidth]
  Cheerfully you talk to me \\
  about waves \\
  and particles

  eclectic

  about forward \\
  and reverse phases

  magnetic

  about tension \\
  and discharge

  electric

  Quietly I listen to you \\
  only with gravity

  weak \\
  within the reach \\
  of your hands
\end{verse}
