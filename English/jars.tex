\poemtitle{}

\settowidth{\versewidth}{You hand me a second jar}

\bigskip

\begin{verse}[\versewidth]
  Without a thought \\
  you hand me a second jar

  Without a thought \\
  I open it and you smile \\
  in profile \\
  at this intimacy \\
  like a wife

  Hidden in plain sight \\
  secret delights \\
  that can't touch \\
  or be open \\
  without a thought
\end{verse}
