\settowidth{\versewidth}{float in a world of silence}

\bigskip

\begin{verse}[\versewidth]
  Among the guests \\
  beaming and laughing \\
  at brilliant jokes\\
  we share a drink\\
  with two ice cubes

  Side by side \\
  fatal icebergs\\
  complicit and beautiful \\
  float in a world of silence

  short-changing the suns\\
  ninety percent
\end{verse}
