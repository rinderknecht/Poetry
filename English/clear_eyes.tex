\settowidth{\versewidth}{where I seek my reflection}

\bigskip

\begin{verse}[\versewidth]
  Those clear eyes

  clearer yet this morning \\
  to have loved another

  A clearer sky

  whose rising sun \\
  left two puddles \\
  where I seek my reflection \\
  amid the dew
\end{verse}
